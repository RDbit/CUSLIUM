const int trigPin = 5;     // D5 - Trigger
const int echoPin = 4;     // D4 - Echo
const int buzzerPin = 3;   // D3 - PWM Buzzer

long duration;
int distance;
int pwmValue;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzerPin, OUTPUT);

  Serial.begin(9600); // Start Serial Monitor
}

void loop() {
  // Send ultrasonic pulse
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);

  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  // Measure time of echo
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  // Set PWM based on distance
  if (distance >= 0 && distance <= 30) {
    pwmValue = 255;  // High vibration
  } else if (distance > 30 && distance <= 60) {
    pwmValue = 200;  // Medium
  } else if (distance > 60 && distance <= 100) {
    pwmValue = 170;  // Low
  } else {
    pwmValue = 0;    // No vibration
  }

  // Apply PWM to buzzer
  analogWrite(buzzerPin, pwmValue);

  // Print data to Serial Monitor
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.print(" cm | PWM: ");
  Serial.println(pwmValue);

  delay(100); // Wait a bit
}
